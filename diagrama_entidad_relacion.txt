Table Gimnasio {
  ID_Gimnasio int [pk]
  Nombre varchar
  Direccion varchar
  Numero_Telefono varchar
}

Table Miembro {
  ID_Miembro int [pk]
  Nombre varchar
  APaterno varchar
  AMaterno varchar
  Num_Telefono varchar
  Direccion varchar
  Fecha_Nac date
  Foto varchar
}

Table Empleado {
  ID_Empleado int [pk]
  ID_Gimnasio int [ref: > Gimnasio.ID_Gimnasio]
  Nombre varchar
  APaterno varchar
  AMaterno varchar
  Num_Telefono varchar
  Direccion varchar
  Fecha_Nac date
  Fecha_Ingreso date
}

Table Rol {
  ID_Rol int [pk]
  Nombre varchar   // Owner, Admin, Empleado
}

Table Usuario {
  ID_Usuario int [pk]
  ID_Empleado int [ref: > Empleado.ID_Empleado]
  ID_Rol int [ref: > Rol.ID_Rol]
  Username varchar
  Password varchar
}

Table Producto {
  ID_Producto int [pk]
  ID_Gimnasio int [ref: > Gimnasio.ID_Gimnasio]
  Nombre varchar
  Cantidad int
  Precio decimal
}

Table Membresia {
  ID_Membresia int [pk]
  Modalidad varchar       // diaria, semanal, mensual
  Precio decimal
  Duracion_Meses int      // ej: mensual=1, trimestral=3
}

Table Promocion {
  ID_Promocion int [pk]
  Nombre varchar
  Descripcion varchar
  Fecha_Inicio date
  Fecha_Fin date
  Meses_Compra int
  Meses_Bonificados int
  Activa boolean
}

Table Pago_Membresia {
  ID_Pago int [pk]
  ID_Miembro int [ref: > Miembro.ID_Miembro]
  ID_Empleado int [ref: > Empleado.ID_Empleado]
  ID_Membresia int [ref: > Membresia.ID_Membresia]
  ID_Promocion int [ref: > Promocion.ID_Promocion, null]
  Fecha_Pago date
  Fecha_Inicio date
  Fecha_Fin date
  Monto decimal
}

Table Acceso_Miembro {
  ID_Acceso int [pk]
  ID_Miembro int [ref: > Miembro.ID_Miembro]
  ID_Gimnasio int [ref: > Gimnasio.ID_Gimnasio]
  ID_Empleado int [ref: > Empleado.ID_Empleado, null] // quien valid√≥ la entrada
  Fecha_Hora datetime
}
